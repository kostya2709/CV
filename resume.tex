\newcommand{\resumeSubheading}[4]{
  \vspace{-1pt}\item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \textbf{#1} & #2 \\
      \textit{\small#3} & \textit{\small #4} \\

  \tikz[baseline]\node[anchor=base,draw=shadecolor!70,rounded corners=0.9ex,inner xsep=1ex,inner ysep =0.55ex,text height=1.4ex,text depth=.25ex]{#1};
  \vspace{-1pt}
